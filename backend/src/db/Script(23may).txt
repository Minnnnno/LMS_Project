CREATE TABLE organisations (
  org_id SERIAL PRIMARY KEY,
  org_name VARCHAR(255) NOT NULL
);

CREATE TABLE users (
  user_id SERIAL PRIMARY KEY,
  first_name VARCHAR(100) NOT NULL,
  last_name VARCHAR(100) NOT NULL,
  email VARCHAR(255) NOT NULL UNIQUE,
  password_hash VARCHAR(255) NOT NULL,
  org_id INT,
  FOREIGN KEY (org_id)
    REFERENCES organisations(org_id)
    ON DELETE RESTRICT
);

CREATE TYPE content_type AS ENUM ('video', 'slides', 'file', 'text');

CREATE TABLE courses (
  course_id SERIAL PRIMARY KEY,
  name VARCHAR(255) NOT NULL,
  org_id INT NOT NULL,
  status VARCHAR(20) NOT NULL DEFAULT 'draft',

  price_cents INT NOT NULL DEFAULT 0,
  currency CHAR(3) NOT NULL DEFAULT 'sgd',
  is_paid BOOLEAN NOT NULL DEFAULT FALSE,

  FOREIGN KEY (org_id)
    REFERENCES organisations(org_id)
    ON DELETE RESTRICT
);

CREATE TABLE enrollments (
  user_id INT NOT NULL,
  course_id INT NOT NULL,
  enrolled_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

  PRIMARY KEY (user_id, course_id),

  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE,

  FOREIGN KEY (course_id)
    REFERENCES courses(course_id)
    ON DELETE CASCADE
);

CREATE TABLE payments (
  payment_id SERIAL PRIMARY KEY,
  user_id INT NOT NULL,
  course_id INT NOT NULL,

  provider VARCHAR(50) NOT NULL DEFAULT 'stripe',

  checkout_session_id VARCHAR(255) UNIQUE,
  payment_ref VARCHAR(255) UNIQUE,

  amount_cents INT NOT NULL,
  currency CHAR(3) NOT NULL DEFAULT 'sgd',

  payment_status VARCHAR(30) NOT NULL DEFAULT 'PENDING',
  paid_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,

  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE,

  FOREIGN KEY (course_id)
    REFERENCES courses(course_id)
    ON DELETE CASCADE
);

CREATE TABLE roles (
  role_id SERIAL PRIMARY KEY,
  description VARCHAR(100) NOT NULL UNIQUE,
  org_id INT,

  FOREIGN KEY (org_id)
    REFERENCES organisations(org_id)
    ON DELETE CASCADE
);

CREATE TABLE user_roles (
  user_id INT NOT NULL,
  role_id INT NOT NULL,

  PRIMARY KEY (user_id, role_id),

  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE,

  FOREIGN KEY (role_id)
    REFERENCES roles(role_id)
    ON DELETE CASCADE
);

CREATE TABLE modules (
  module_id SERIAL PRIMARY KEY,
  course_id INT NOT NULL,
  instructor_id INT,
  title VARCHAR(255) NOT NULL,
  position INT NOT NULL,

  FOREIGN KEY (course_id)
    REFERENCES courses(course_id)
    ON DELETE CASCADE,

  FOREIGN KEY (instructor_id)
    REFERENCES users(user_id)
    ON DELETE SET NULL
);

CREATE TABLE module_contents (
  module_content_id SERIAL PRIMARY KEY,
  module_id INT NOT NULL,
  content_type content_type NOT NULL,
  title VARCHAR(255),
  content_url VARCHAR(2083),
  cloudinary_public_id VARCHAR(255),
  content_body TEXT,
  position INT NOT NULL,

  FOREIGN KEY (module_id)
    REFERENCES modules(module_id)
    ON DELETE CASCADE
);

CREATE TABLE lesson_progress (
  user_id INT NOT NULL,
  module_id INT NOT NULL,
  completed_at TIMESTAMPTZ,

  PRIMARY KEY (user_id, module_id),

  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE,

  FOREIGN KEY (module_id)
    REFERENCES modules(module_id)
    ON DELETE CASCADE
);

CREATE TABLE assignments (
  assignment_id SERIAL PRIMARY KEY,
  course_id INT NOT NULL,
  title VARCHAR(255) NOT NULL,
  description TEXT,
  due_date TIMESTAMPTZ,
  max_score DECIMAL(5, 2),

  FOREIGN KEY (course_id)
    REFERENCES courses(course_id)
    ON DELETE CASCADE
);

CREATE TABLE submissions (
  submission_id SERIAL PRIMARY KEY,
  assignment_id INT NOT NULL,
  user_id INT NOT NULL,
  submitted_at TIMESTAMPTZ NOT NULL DEFAULT CURRENT_TIMESTAMP,
  submission_text TEXT,
  file_url VARCHAR(2083),
  cloudinary_public_id VARCHAR(255),
  score DECIMAL(5, 2),
  feedback TEXT,

  FOREIGN KEY (assignment_id)
    REFERENCES assignments(assignment_id)
    ON DELETE CASCADE,

  FOREIGN KEY (user_id)
    REFERENCES users(user_id)
    ON DELETE CASCADE
);
